#Change order of type checking

void _setValue(int base, int offset, value) {
    if (value is double) {
      value = value.toInt();
    }

    if (value is int) {
      Unsafe.writeInt32(base, offset, value);
    } else {
      super._setValue(base, offset, value);
    }
  }
